\begin{recipe}{Schaschliktopf}{Reis}{3 h}{4}

\begin{tcolorbox}[colback=cookSecondary!60, colframe=white, left=4mm, right=4mm, boxrule=0pt, top=2mm,bottom=2mm, sharp corners]
\begin{minipage}{0.28\textwidth}
% \includegraphics[width=\linewidth]{}
\end{minipage}\hfill
\begin{minipage}{0.68\textwidth}
\textit{Schaschlik aus dem Topf. Dazu passt: Reis, Kartoffeln, Baguette.}
\end{minipage}
\end{tcolorbox}

\vspace{6pt}

\begin{tcolorbox}[colback=white,colframe=black!5,boxrule=0.5pt,arc=3mm]
\textbf{Zutaten}
\begin{itemize}[leftmargin=*,noitemsep]
\item 500 g Schweinefleisch
\item 1 Stück geräuchertes Wammerl
\item 2 Zwiebeln
\item 2 Knoblauchzehen
\item 3 Paprika
\item 700 ml passierte Tomaten
\item 100 ml Schaschliksauce
\item 2 EL Worchestersauce
\item 1 TL Senf
\item 1 TL Honig
\item Paprika edelsü{\ss}, Curry, Gemüse-/Rinderbrühe, Rauchsalz, Barbecuegewürz, Chili
\end{itemize}
\end{tcolorbox}

\vspace{4pt}

\begin{tcolorbox}[colback=white,colframe=black!5,boxrule=0.5pt,arc=3mm]
\textbf{Zubereitung}
\begin{itemize}[leftmargin=*,noitemsep]
\item Fleisch in Würfel schneiden.
\item Zwiebeln achteln \& Paprika grob würfeln.
\item Wammerl in Butterschmalz anbraten, herausnehmen.
\item Schweinefleisch anbraten, salzen \& pfeffern und anschlie{\ss}end herausnehmen.
\item Zwiebeln anbraten, Paprika \& Knoblauch hinzugeben und mitbraten.
\item Fleisch wieder hinzugeben und kräftig würzen.
\item Passierte Tomaten \& Schaschliksauce dazugeben und mit Wasser aufgie{\ss}en, bis alles bedeckt ist.
\item Worchestersauce, Senf \& Honig einrühren.
\item Mind. 2h köcheln lassen.
\end{itemize}
\end{tcolorbox}

\end{recipe}
