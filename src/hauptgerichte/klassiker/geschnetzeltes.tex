\begin{recipe}{Geschnetzeltes in Rahm Champignons}{Reis}{1 h}{2}

\begin{tcolorbox}[colback=cookSecondary!60, colframe=white, left=4mm, right=4mm, boxrule=0pt, top=2mm,bottom=2mm, sharp corners]
\begin{minipage}{0.28\textwidth}
% \includegraphics[width=\linewidth]{src/images/FCF-98.jpg}
\end{minipage}\hfill
\begin{minipage}{0.68\textwidth}
\textit{Geschnetzeltes in Rahm Champignons. Dazu passt: Möhren-Erbsen Gemüse, Reis, Kartoffelrösti, Spätzle.}
\end{minipage}
\end{tcolorbox}

\vspace{6pt}

\begin{tcolorbox}[colback=white,colframe=black!5,boxrule=0.5pt,arc=3mm]
\textbf{Zutaten}
\begin{itemize}[leftmargin=*,noitemsep]
\item 400 g (Hähnchen-) Schnitzel
\item 250 g braune Champignons
\item 1 Zwiebel
\item 100 ml trockener Wei{\ss}wein
\item 100 ml Sahne
\item 250 ml Rinderbrühe
\item 1 EL Mehl
\item 1 EL Sojasauce
\item 1 TL Tomatenmark
\item 1 TL scharfer Senf
\item Pfefferkörner bunt
\item frische Petersilie
\end{itemize}
\end{tcolorbox}

\vspace{4pt}

\begin{tcolorbox}[colback=white,colframe=black!5,boxrule=0.5pt,arc=3mm]
\textbf{Zubereitung}
\begin{itemize}[leftmargin=*,noitemsep]
\item Fleisch in Streifen schneiden.
\item Fleischstreifen in Butterschmalz anbraten, würzen \& aus der Pfanne nehmen.
\item Pilze schneiden \& anbraten.
\item Anschlie{\ss}end die gewürfelte Zwiebeln hinzugeben \& mit Tomatenmark rösten.
\item Mit Mehl stäuben \& Wein ablöschen. Im Anschluss reduzieren lassen.
\item Brühe, Sojasauce, Senf \& Pfefferkörner hinzugeben. Aufkochen lassen.
\item Fleisch wieder zurückgeben \& etwas kochen lassen.
\item Zum Abschluss Sahne \& Petersilie unterrühren. 
\end{itemize}
\end{tcolorbox}

\end{recipe}
