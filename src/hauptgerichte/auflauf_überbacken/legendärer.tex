\begin{recipe}{Legendärer Nudelauflauf}{Bowl}{1 h}{5}

\begin{tcolorbox}[colback=cookSecondary!60, colframe=white, left=4mm, right=4mm, boxrule=0pt, top=2mm,bottom=2mm, sharp corners]
\begin{minipage}{0.28\textwidth}
% \includegraphics[width=\linewidth]{src/images/spaghetti.jpg}
\end{minipage}\hfill
\begin{minipage}{0.68\textwidth}
\textit{Der legendäre Nudelauflauf á la Monte. Johannes beherrscht ihn in Perfektion.}
\end{minipage}
\end{tcolorbox}

\vspace{6pt}

\begin{tcolorbox}[colback=white,colframe=black!5,boxrule=0.5pt,arc=3mm]
\textbf{Zutaten}
\begin{itemize}[leftmargin=*,noitemsep]
\item 500g Nudeln (Fusilli)
\item 500g Hackfleisch
\item 1 rote Zwiebel
\item 1 Paprika
\item Champignons
\item Mais
\item Maggi (Ofen-Spirelli alla Mamma) oder Knorr (Ofen-Makkaroni alla Mamma)
\item Creme Fraiche
\item Geriebener Käse
\item Mozzarella
\item Jalapeños
\end{itemize}
\end{tcolorbox}

\vspace{4pt}

\begin{tcolorbox}[colback=white,colframe=black!5,boxrule=0.5pt,arc=3mm]
\textbf{Zubereitung}
\begin{itemize}[leftmargin=*,noitemsep]
\item Nudeln ca. 5min vorkochen.
\item Hackfleisch scharf anbraten.
\item Geschnittene Zwiebel, Paprika, Champignons \& Mais dazugeben.
\item Mit Wasser aufgie{\ss}en, Maggi \& Creme Fraiche zugeben und kurz köcheln lassen.
\item Nudeln dazugeben, vermischen \& in eine Auflaufform geben.
\item Käse, Mozzarella \& Jalapeños darüber verteilen.
\item Bei \ang{180}C ca. 30-40 min in den Backofen.
\item Wahlweise mit Hela Gewürzketchup servieren.
\end{itemize}
\end{tcolorbox}

\end{recipe}
