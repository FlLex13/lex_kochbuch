\begin{recipe}{Spaghetti Carbonara}{Pasta}{20 Min}{4}

\begin{tcolorbox}[colback=cookSecondary!60, colframe=white, left=4mm, right=4mm, boxrule=0pt, top=2mm,bottom=2mm, sharp corners]
\begin{minipage}{0.28\textwidth}
% \includegraphics[width=\linewidth]{src/images/spaghetti.jpg}
\end{minipage}\hfill
\begin{minipage}{0.68\textwidth}
\textit{Klassisches italienisches Pastagericht mit Eiern und Pecorino.}
\end{minipage}
\end{tcolorbox}

\vspace{6pt}

\begin{tcolorbox}[colback=white,colframe=black!5,boxrule=0.5pt,arc=3mm]
\textbf{Zutaten}
\begin{itemize}[leftmargin=*,noitemsep]
\item 400 g Spaghetti
\item 150 g Pancetta oder Speck
\item 3 Eier
\item 80 g Pecorino oder Parmesan, gerieben
\item Pfeffer, Salz
\end{itemize}
\end{tcolorbox}

\vspace{4pt}

\begin{tcolorbox}[colback=white,colframe=black!5,boxrule=0.5pt,arc=3mm]
\textbf{Zubereitung}
\begin{itemize}[leftmargin=*,noitemsep]
\item Spaghetti in reichlich Salzwasser al dente kochen.
\item Pancetta in Würfel schneiden und knusprig braten.
\item Eier mit Käse und schwarzem Pfeffer verquirlen.
\item Nudeln abgießen, etwas Kochwasser auffangen. Nudeln sofort mit Pancetta und Eimischung vermengen (Hitze der Pasta stockt die Eier). Bei Bedarf etwas Kochwasser zugeben.
\item Sofort servieren.
\end{itemize}
\end{tcolorbox}

\end{recipe}
