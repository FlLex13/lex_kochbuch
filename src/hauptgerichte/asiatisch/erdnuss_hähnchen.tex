\begin{recipe}{Knuspriges Hähnchen mit Erdnusssauce}{Asiatisch}{1 h}{2}

\begin{tcolorbox}[colback=cookSecondary!60, colframe=white, left=4mm, right=4mm, boxrule=0pt, top=2mm,bottom=2mm, sharp corners]
\begin{minipage}{0.28\textwidth}
% \includegraphics[width=\linewidth]{src/images/spaghetti.jpg}
\end{minipage}\hfill
\begin{minipage}{0.68\textwidth}
\textit{Hähnchen in knuspriger Panko-Panade mit cremiger Erdnusssauce. Dazu passt: Reis \& Gemüse.}
\end{minipage}
\end{tcolorbox}

\vspace{6pt}

\begin{tcolorbox}[colback=white,colframe=black!5,boxrule=0.5pt,arc=3mm]
\textbf{Zutaten}
\begin{itemize}[leftmargin=*,noitemsep]
\item 1 Tasse Reis
\item 2 Sü{\ss}kartoffeln
\item 2 Karotten
\item 1 Brokkoli
\item 2 Hähnchenbrustfilets
\item 1 Ei
\item Panko-Panade
\item Erdnusssauce:
    \begin{itemize}
        \item 65g Erdnussbutter
        \item 2 EL Sojasauce
        \item 1 EL Reisessig
        \item 1 EL Sesamöl
        \item 2 TL Sriracha
        \item 1 TL Honig
    \end{itemize}
\end{itemize}
\end{tcolorbox}

\vspace{4pt}

\begin{tcolorbox}[colback=white,colframe=black!5,boxrule=0.5pt,arc=3mm]
\textbf{Zubereitung}
\begin{itemize}[leftmargin=*,noitemsep]
\item Sü{\ss}kartoffeln \& Karotten in Scheiben schneiden und mit Öl marinieren \& würzen.
\item Bei \ang{180}C ca. 25 min in den Backofen.
\item Die letzten 10 min die Brokkoliröschen dazugeben (Der Strunk kann geschält und in Scheiben geschnitten auch verwendet werden).
\item Reis kochen.
\item Hähnchenbrust salzen \& pfeffern. Danach durch die Panierstra{\ss}e:
    \begin{itemize}
        \item Speisestärke.
        \item Gequirltes Ei mit ein wenig Sojasauce.
        \item Panko-Panade.
    \end{itemize}
\item In hei{\ss}em Öl frittieren \& auf Küchentuch abtropfen lassen.
\item Die Zutaten für die Erdnusssauce verrühren \& mit hei{\ss}em Wasser aufgie{\ss}en, bis die gewünschte Konsistenz erreicht ist.
\end{itemize}
\end{tcolorbox}

\end{recipe}
